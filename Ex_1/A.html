Практика №1.

Задача А. 

        #include <algorithm>
#include <stdio.h>
using namespace std;

// Иначе студия ругается на scanf \ printf
#pragma warning(disable:4996)

int main()
{
    int n;
    scanf("%d", &n);
    int max_number = 1 << n;
    int max_position = 1 << (n - 1);
    for (int i = 0; i<max_number; ++i)
    {
        for (int maska = max_position; maska>0; maska >>= 1)
        {
            if ((i&maska) == 0)
            {
                printf("+");
            }
            else
            {
                printf("-");
            }
        }
        printf("\n");
    }
    return 0;
}
